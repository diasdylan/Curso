git branch

## Creates a new branch and changes to it
git checkout -b 'August'

## Creates a new branch
git branch 'August'

## Git Add all TRACKED modified files and directores 
git add -u

## Undo last commit but leave file staged
git reset --soft head~1

## Commit with header and description
git commit -m 'header text' -m 'description text'

## Check last commits
git log

## Check last commit, short format
git log --oneline
